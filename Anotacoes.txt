"""Modelo do sistema PГЄndulo Invertido."""

from control import TransferFunction
from .base_system import BaseSystem

class PendulumSystem(BaseSystem):
    """Representa o sistema PГЄndulo Simples Invertido."""
    
    def __init__(self, m: float, g: float, L: float, b: float):
        """
        Args:
            m: Massa da haste (kg)
            g: Gravidade (m/sВІ)
            L: Comprimento da haste (m)
            b: Coeficiente de viscosidade (NВ·mВ·s/rad)
        """
        super().__init__("Inverted-Pendulum")
        self.m = m
        self.g = g
        self.L = L
        self.b = b
        self._update_transfer_function()
    
    def _update_transfer_function(self):
        """Atualiza a funГ§ГЈo de transferГЄncia."""
        numerator = [3 * self.m / self.L**2]
        denominator = [1, 3 * self.b / (self.m * self.L**2), -3 * self.g / (2 * self.L)]
        self._transfer_function = TransferFunction(numerator, denominator)
    
    def get_transfer_function(self) -> TransferFunction:
        """Retorna H(s) do pГЄndulo invertido."""
        return self._transfer_function
    
    def get_parameters_dict(self) -> dict:
        """Retorna parГўmetros do sistema."""
        I = self.m * self.L**2 / 3
        return {
            'm': self.m,
            'g': self.g,
            'L': self.L,
            'b': self.b,
            'I': I
        }
    
    def calculate_min_torque(self) -> float:
        """Calcula torque mГ­nimo para dar uma volta completa."""
        return self.m * self.g * self.L / 2
